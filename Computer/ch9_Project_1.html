Project 1 - Wish Me

This code uses the datetime module to get the current time. Then, it compares the hour component of the current time to determine whether it is morning (before 12), afternoon (between 12 and 18), or evening (after 18). Finally, it prints the appropriate greeting based on the time. 
import datetime

current_time = datetime.datetime.now().time()

if current_time.hour < 12:
    print("Good Morning!")
elif 12 <= current_time.hour < 18:
    print("Good Afternoon!")
else:
    print("Good Evening!")

Project 2 - Check Email

In this code, we define the is_valid_email function that uses a regular expression pattern to check if an email is valid. The validate_emails function iterates over the dictionary items, checks each email using the is_valid_email function, and prints whether it is valid or invalid along with the corresponding name. 
import re

def is_valid_email(email):
    # Regular expression pattern to validate email format
    pattern = r'^[\w\.-]+@[\w\.-]+\.\w+$'
    return re.match(pattern, email)

def validate_emails(dictionary):
    for name, email in dictionary.items():
        if is_valid_email(email):
            print(f"Valid email found for {name}: {email}")
        else:
            print(f"Invalid email found for {name}: {email}")

# Example dictionary with names and emails
email_dict = {
    "John": "john@example.com",
    "Emma": "emma.example.com",
    "Michael": "michael@example.com",
    "Sophia": "sophia@example.com"
}

# Validate the emails in the dictionary
validate_emails(email_dict)

Project 3 - Number Guessing Game

In this game, the program generates a random number between 1 and 100 using the random.randint() function. The player has to guess the number, and the program provides feedback if the guess is too low or too high. It keeps track of the number of attempts until the correct number is guessed. Once the correct number is guessed, it displays a congratulatory message along with the number of attempts. 
import random

def number_guessing_game():
    secret_number = random.randint(1, 100)
    attempts = 0

    print("Welcome to the Number Guessing Game!")
    print("I have chosen a number between 1 and 100. Can you guess it?")

    while True:
        guess = int(input("Enter your guess: "))
        attempts += 1

        if guess == secret_number:
            print(f"Congratulations! You guessed the correct number in {attempts} attempts.")
            break
        elif guess < secret_number:
            print("Too low. Try again.")
        else:
            print("Too high. Try again.")

    print("Thank you for playing!")

# Run the game
number_guessing_game()

Project 4 - To-Do List Application

In this code, the To-Do List Application allows the user to add tasks, remove tasks, and view the tasks in the to-do list. It uses a simple while loop to display a menu and prompts the user for their choice. Based on the selected option, the corresponding function is called to perform the desired operation. 
# To-Do List Application

# Empty to-do list
todo_list = []

# Function to add a task to the to-do list
def add_task(task):
    todo_list.append(task)
    print("Task added successfully.")

# Function to remove a task from the to-do list
def remove_task(task):
    if task in todo_list:
        todo_list.remove(task)
        print("Task removed successfully.")
    else:
        print("Task not found.")

# Function to view all tasks in the to-do list
def view_tasks():
    if todo_list:
        print("Tasks:")
        for task in todo_list:
            print("- " + task)
    else:
        print("No tasks in the to-do list.")

# Main program loop
while True:
    print("\n==== To-Do List Application ====")
    print("1. Add Task")
    print("2. Remove Task")
    print("3. View Tasks")
    print("4. Exit")

    choice = input("Enter your choice (1-4): ")

    if choice == "1":
        task = input("Enter the task: ")
        add_task(task)
    elif choice == "2":
        task = input("Enter the task to remove: ")
        remove_task(task)
    elif choice == "3":
        view_tasks()
    elif choice == "4":
        print("Exiting the program.")
        break
    else:
        print("Invalid choice. Please try again.")

Project 6 - Dice Rolling Simulator

In this code, the roll_dice() function generates a random number between 1 and 6, simulating the roll of a dice. The play_again() function prompts the user to play again or quit. The main() function is the entry point of the program, where the dice rolling simulation takes place.

When you run the code, it will ask you to press Enter to roll the dice. After rolling the dice, it will display the result. Then, it will ask if you want to roll the dice again. If you choose to continue, it will roll the dice again; otherwise, it will exit the program. 
import random

def roll_dice():
    min_value = 1
    max_value = 6
    return random.randint(min_value, max_value)

def play_again():
    while True:
        choice = input("Do you want to roll the dice again? (y/n): ")
        if choice.lower() == 'y':
            return True
        elif choice.lower() == 'n':
            return False
        else:
            print("Invalid choice. Please enter 'y' or 'n'.")

def main():
    print("Welcome to the Dice Rolling Simulator!")

    while True:
        input("Press Enter to roll the dice...")
        dice_value = roll_dice()
        print("The dice rolled and you got:", dice_value)

        if not play_again():
            break

    print("Thank you for playing!")

if __name__ == '__main__':
    main()

Project 8 - Password Generator

In this code, the generate_password function takes an optional parameter length (default value is 8) that specifies the length of the password. The function generates a password by randomly selecting characters from a combination of uppercase letters, lowercase letters, digits, and punctuation marks. 
import random
import string

def generate_password(length=8):
    characters = string.ascii_letters + string.digits + string.punctuation
    password = ''.join(random.choice(characters) for _ in range(length))
    return password

# Example usage
password = generate_password()
print("Generated Password:", password)

Project 9 - Quiz Game

In this example, the Question class represents a single quiz question with its choices and correct answer. The Quiz class manages the flow of the quiz, keeps track of the score, and interacts with the user. You can modify the questions, choices, and answers in the questions list to customize the quiz according to your preferences. 
class Question:
    def __init__(self, question, choices, answer):
        self.question = question
        self.choices = choices
        self.answer = answer

    def check_answer(self, user_answer):
        return user_answer.lower() == self.answer.lower()


class Quiz:
    def __init__(self, questions):
        self.questions = questions
        self.score = 0
        self.current_question_index = 0

    def get_current_question(self):
        return self.questions[self.current_question_index]

    def display_question(self):
        question = self.get_current_question()
        print(question.question)
        for index, choice in enumerate(question.choices):
            print(f"{index + 1}. {choice}")

    def get_user_answer(self):
        while True:
            try:
                user_input = int(input("Enter your answer (1-4): "))
                if 1 <= user_input <= 4:
                    return user_input
                else:
                    print("Invalid input. Please enter a number between 1 and 4.")
            except ValueError:
                print("Invalid input. Please enter a number.")

    def check_user_answer(self, user_answer):
        question = self.get_current_question()
        if question.check_answer(user_answer):
            self.score += 1
            print("Correct answer!\n")
        else:
            print("Wrong answer!\n")
        self.current_question_index += 1

    def display_score(self):
        total_questions = len(self.questions)
        print(f"You scored {self.score} out of {total_questions}.\n")

    def start_quiz(self):
        print("Welcome to the Quiz Game!\n")
        while self.current_question_index < len(self.questions):
            self.display_question()
            user_answer = self.get_user_answer()
            self.check_user_answer(user_answer)
        self.display_score()


# Define your quiz questions
questions = [
    Question("What is the capital of France?", ["London", "Paris", "Madrid", "Rome"], "Paris"),
    Question("Which planet is known as the Red Planet?", ["Mars", "Venus", "Jupiter", "Mercury"], "Mars"),
    Question("Who painted the Mona Lisa?", ["Pablo Picasso", "Leonardo da Vinci", "Vincent van Gogh", "Michelangelo"],
             "Leonardo da Vinci"),
    Question("What is the largest ocean in the world?", ["Atlantic Ocean", "Arctic Ocean", "Indian Ocean", "Pacific Ocean"],
             "Pacific Ocean"),
]

# Create a new quiz instance and start the quiz
quiz = Quiz(questions)
quiz.start_quiz()
